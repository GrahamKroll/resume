                CREATE TABLE IF NOT EXISTS `job_history` (
                `EMPID` varchar(6) NOT NULL,
                `STARTDATE` date NOT NULL,
                `ENDDATE` date NOT NULL,
                `JOBID` varchar(10) NOT NULL,
                `DEPTID` varchar(4) DEFAULT NULL,
                PRIMARY KEY (`EMPID`,`STARTDATE`),
                CONSTRAINT job_history_empid_fk FOREIGN KEY (empid) REFERENCES Employee (empid) ON DELETE CASCADE,
                CONSTRAINT job_history_jobid_fk FOREIGN KEY (jobid) REFERENCES Job (jobid) ON DELETE CASCADE,
                CONSTRAINT job_history_deptid_fk FOREIGN KEY (deptid) REFERENCES Department (deptid) ON DELETE CASCADE
                );
